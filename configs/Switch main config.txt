#Préparation des vlan et leurs IPs
#vlan 1 (DB)

name DB
exit

switchport mode access
switchport access vlan 1
no sh
exit
interface vlan 1
ip address 10.1.32.1 255.255.224.0
ip helper-address 10.1.64.2
no sh


#vlan 2 (servers)

vlan 2
name servers
exit

switchport mode access
switchport access vlan 2
no sh
exit
interface vlan 2
ip address 10.1.64.1 255.255.224.0
no sh
exit


#vlan 3 (Admins)
